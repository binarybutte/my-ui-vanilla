Math.max()

  • https://developer.mozilla.org/ko/docs/Web/JavaScript/Reference/Global_Objects/Math/max

매개변수로 주어진 숫자 중 가장 큰 수를 반환하거나, 매개변수가 없을 경우 -Infinity를 반환

const array1 = [1, 3, 2];
Math.max(...array1) // 3
Math.max(1, 3, 2) // 3
Math.max(-1, -3, -2) // -1

// ↓↓ 배열에서 이전 인덱스 구하기 예제. 0보다 작을 경우를 방지
const array2 = [0, 1, 2, 3, 4];
const currentScene = 1; // 변할 수 있음
const idx = Math.max(0, Number(currentScene) - 1);
const prevResult = array2[idx];
or
const prevObjs = (array2[currentScene - 1] ?? array2[0])?.objects; // ?.objects 뺴도 됨
or
const prevObjs = (currentScene > 0 ? array2[currentScene - 1] : array2[0])?.objects;

// ↓↓ 배열에서 다음 인덱스 구하기 예제. 배열 갯수보다 클 경우를 방지
// 범위를 벗어나면 0 또는 length-1에 고정
const array3 = [0, 1, 2, 3, 4];
const currentScene = 2; // 변할 수 있음
const next = Number(currentScene) + 1;
const maxIdx = array3.length - 1;
const idx = Math.max(0, Math.min(maxIdx, next));
or
const clamp = (n, min, max) => Math.max(min, Math.min(max, n));
const idx = clamp(Number(currentScene) + 1, 0, array3.length - 1);

// 범위를 벗어나면 처음/끝으로 순환
const len = array3.length;
const idx = len ? ((Number(currentScene) + 1) % len + len) % len : 0;
