EventListener 설정 옵션

{ once: true }

once: true를 사용하면 이벤트 리스너가 한 번만 실행되고 자동으로 삭제됨. onclick 인라인으로 적용할 때에는 함수 내부의 클릭이벤트에만 걸어줘도 된다.
버튼 클릭 시 한 번만 실행되는 로직에 유용. 

btn1.addEventListener("click", function btnclickHandler() {
	btn2.addEventListener("click", function () { console.log("버튼2 클릭!"); }, { once: true });
}, { once: true });

<button id="btn1" onclick="btnClickHandler()">버튼1 클릭</button>
<button id="btn2">버튼2 클릭</button>

function btnClickHandler() {
	const btn2 = document.getElementById("btn2");
	btn2.addEventListener("click", function () {
		console.log("버튼2 클릭!");
	}, { once: true }); 
}

{ passive: false }

passive 속성이 false인 경우에 touchstart, touchmove와 같은 이벤트가 발생하면 preventDefault를 이용하여 실제 이벤트 자체를 막을 수 있다.
모바일에서 터치 스크롤을 제어하고 싶을 때 preventDefault()가 필요한 경우 사용. 

touchBlock: function () {
	const touchBtns = document.querySelectorAll('.btns');
	touchBtns.forEach(function (item) {
		item.addEventListenter('touchstart', e => {
			e.preventDefault;
			e.stopImmediateProoagation();
		}, {passive: false})
	})
}
