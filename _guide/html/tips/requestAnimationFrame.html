let lastTime = 0;
function messageChange (timestamp) {
	const currentTime = new Date().getTime();
	const message01 = $('.m01') // 문구 1
	const message02 = $('.m02') // 문구 2
	if ( currentTime - lastTime > 3000 ) {
		message02.addClass('active');
		message01.removeClass('active')
	}
	if ( currentTime - lastTime > 6000 ) {
		message01.addClass('active');
		message02.removeClass('active')
		lastTime = currentTime // 재시작
	}
	repeat = requestAnimationFrame(messageChange)
	repeat;
}
